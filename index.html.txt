    <script>
        function goToPage(num) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById('page' + num).classList.add('active');
            if(num === 3) {
                const video = document.getElementById('bdayVideo');
                video.onended = () => {
                    document.getElementById('poemContainer').style.display = 'block';
                };
            }
        }

        // Page 1 Logic
        function growTree() {
            document.getElementById('seed').style.display = 'none';
            document.getElementById('tree').style.display = 'block';
            const blossoms = document.getElementById('blossoms');
            for(let i=0; i<15; i++) {
                let h = document.createElement('div');
                h.className = 'mini-heart';
                h.innerHTML = '‚ô•';
                h.style.color = i % 2 === 0 ? '#ff85a1' : '#ffe162';
                blossoms.appendChild(h);
            }
            blossoms.classList.add('show-blossoms');
            setTimeout(() => {
                document.getElementById('heartBtn').style.display = 'block';
            }, 3000);
        }

        // Page 2 Fireworks logic (Simplified Canvas)
        function launchFireworks() {
            document.getElementById('rocket').style.display = 'none';
            const wish = document.getElementById('bdayWish');
            wish.style.opacity = '1';
            
            // Trigger annoying button movement
            const giftBtn = document.getElementById('giftBtn');
            giftBtn.style.display = 'block';
            
            giftBtn.onmouseover = () => {
                giftBtn.style.left = Math.random() * 80 + '%';
                giftBtn.style.top = Math.random() * 80 + '%';
            };
        }

        // Add drifting background hearts for Page 1
        setInterval(() => {
            const heart = document.createElement('div');
            heart.innerHTML = '‚ô•';
            heart.style.position = 'fixed';
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.top = '100vh';
            heart.style.color = Math.random() > 0.5 ? '#ff85a1' : '#ffe162';
            heart.style.fontSize = Math.random() * 20 + 10 + 'px';
            heart.style.transition = 'all 4s linear';
            heart.style.zIndex = '-1';
            document.body.appendChild(heart);
            setTimeout(() => {
                heart.style.top = '-10vh';
                heart.style.opacity = '0';
            }, 100);
            setTimeout(() => heart.remove(), 4000);
        }, 300);
    </script>
